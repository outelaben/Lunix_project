#!/bin/bash
 
function verif
{
for ((i=1;i<=$#;i++)) do
{
    local ereur=0
    if test ${!i} == "("
    then
        echo "para ouvrante";
        elif   test ${!i} = "+"
        then
            echo "somme";
        elif  test ${!i} = "-"
        then
            echo "soustracion";
        elif  test ${!i} = "X"
        then
            echo "Muliplication";
 
        elif test ${!i} = "/"
        then
            echo "Division";
        elif  test ${!i} = ")"
        then
            echo "para fermante";
        else
            echo "entier";
    fi
}
done
return 1;
}
